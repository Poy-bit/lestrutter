#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in uint material;

out vec4 vertexColor;
uniform mat4 cammat;
uniform mat4 projection;

uniform vec3 material_color[2];

void main() {
    gl_Position = projection * cammat * vec4(aPos, 1.0);
    vertexColor = vec4(material_color[material], 1.0);
}
